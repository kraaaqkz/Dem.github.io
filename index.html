# =============================================================================
# ЛИНЕЙНАЯ РЕГРЕССИЯ: ПРОГНОЗ БАЛЛОВ ПО ЧАСАМ ПОДГОТОВКИ
# Дата: 14 марта 2026
# GitHub: github.com/ваш-логин/linear-regression-project
# =============================================================================

# 📦 ИМПОРТ БИБЛИОТЕК
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_absolute_error, mean_squared_error

# ⚙️ НАСТРОЙКИ
plt.style.use('seaborn-v0_8-whitegrid')
FILE_PATH = "task_regression.csv"

# =============================================================================
# 1. ЗАГРУЗКА ДАННЫХ
# =============================================================================
dataset = pd.read_csv(FILE_PATH, sep=';', decimal=',', encoding='utf-8')
print(f"✅ Загружено: {len(dataset)} записей")

# =============================================================================
# 2. ПОДГОТОВКА: Фичи (X) и Цель (Y)
# =============================================================================
X = dataset[['Часы']]           # Признак: часы подготовки
Y = dataset['Полученные. Баллы'] # Цель: полученные баллы

# =============================================================================
# 3. РАЗДЕЛЕНИЕ: 80% обучение, 20% тест
# =============================================================================
X_train, X_test, Y_train, Y_test = train_test_split(
    X, Y, test_size=0.2, random_state=42, shuffle=True
)

# =============================================================================
# 4. ОБУЧЕНИЕ МОДЕЛИ
# =============================================================================
regressor = LinearRegression()
regressor.fit(X_train, Y_train)

print("\n📊 Параметры модели:")
print(f"   Коэффициент: {regressor.coef_[0]:.4f}")
print(f"   Intercept: {regressor.intercept_:.4f}")
print(f"   Уравнение: Баллы = {regressor.coef_[0]:.2f}×Часы + {regressor.intercept_:.2f}")

# =============================================================================
# 5. ПРОГНОЗ И МЕТРИКИ
# =============================================================================
y_pred = regressor.predict(X_test)

mae = mean_absolute_error(Y_test, y_pred)
rmse = np.sqrt(mean_squared_error(Y_test, y_pred))
r2 = regressor.score(X_test, Y_test)

print(f"\n📈 Метрики:")
print(f"   MAE: {mae:.4f} (средняя ошибка в баллах)")
print(f"   RMSE: {rmse:.4f}")
print(f"   R²: {r2:.4f} ({r2*100:.1f}% объяснённой дисперсии)")

# =============================================================================
# 6. ФУНКЦИЯ ДЛЯ САЙТА: прогноз по часам
# =============================================================================
def predict_score(hours):
    """Прогноз баллов для нового студента"""
    df = pd.DataFrame({'Часы': [hours]})
    return round(regressor.predict(df)[0], 2)

# Пример использования:
for h in [1, 3, 5, 7, 10]:
    print(f"   {h} ч. → {predict_score(h)} балл(ов)")

# =============================================================================
# 7. СОХРАНЕНИЕ МОДЕЛИ (для деплоя)
# =============================================================================
import joblib
joblib.dump(regressor, 'model.pkl')
print("\n💾 Модель сохранена: model.pkl")

# =============================================================================
# 8. ГРАФИК (опционально)
# =============================================================================
plt.figure(figsize=(8, 5))
plt.scatter(X, Y, alpha=0.6, label='Данные')
plt.plot(X, regressor.predict(X), color='red', label='Регрессия')
plt.xlabel('Часы'); plt.ylabel('Баллы')
plt.legend(); plt.grid(alpha=0.3)
plt.savefig('regression_plot.png', dpi=300, bbox_inches='tight')
plt.show()
